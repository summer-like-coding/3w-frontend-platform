                                # 题目描述

                                统计一个数字target在排序数组中出现的次数。

                                # 题目思路

                                1. 使用二分查找的方法，找到target，然后再左右两边分别找到target的边界。
                                  function search(nums: number[], target: number): number {
  // 双指针，二分查找
  let left = 0;
  let right = nums.length - 1;
  while (left <= right) {
    let mid = Math.floor((left + right) / 2);
    // 如果找到了，那么就向左右两边扩散
    if (nums[mid] === target) {
      let count = 1;
      let i = mid - 1;
      let j = mid + 1;
      // 向左边扩散
      while (nums[i] === target) {
        count++;
        i--;
      }
      // 向右边扩散
      while (nums[j] === target) {
        count++;
        j++;
      }
      return count;
    }
    // 如果没找到，那么就继续二分查找
    if (nums[mid] > target) {
      right = mid - 1;
    }
    if (nums[mid] < target) {
      left = mid + 1;
    }
  }
}
